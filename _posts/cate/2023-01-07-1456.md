#include <iostream>

#include <math.h>

int main() {
	long min, max;              // ë´„

	long A[10000001];           // ë´„

	for (int i = 2; i < 10000001; i++) {

			long temp = A[i];
			while ((double)A[i] <= (double)max / (double)temp) {       // ë´„
				if ((double)A[i] >= (double)min / (double)temp)        // ë´„

		}
	}
